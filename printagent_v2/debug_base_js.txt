function utilGetIndexSelect(elem){
	var val = 0;
	if (elem) {
		for (var i = 0; i < elem.length; i++) {
			if ( elem.options[i].selected ) {
				val = i+1;
				break;
			}
		}
	}
	return val;
}

function utilSetIndexSelect(elem,ordinal){
	if (elem) {
		if (elem.length + 1 >= ordinal) {
			elem.options[ordinal-1].selected = SET;
		}
	}
}

function utilGetInputValue(form,name) {
	var el;
	var val = EMPTY;

	if (form) {
		for (var i=0; i < form.length; i++) {
			el = form.elements[i];

			if (el.name == name) {
				if (el.type == "text" || el.type == "password" || el.type == "file" || el.type == "hidden") {
					val = el.value;
					break;
				}
				else if (el.type == "radio") {
					if (el.checked) {
						val = el.value;
						break;
					}
				}
				else if (el.type == "select-one") {
					for(var j=0; j < el.length; j++) {
						if (el.options[j].selected) {
							val = el.options[j].value;
							break;
						}
					}
					break;
				}
				else if (el.type == "select-multiple") {
					for(var j=0; j < el.length; j++) {
						if (el.options[j].selected) {
							if (val != EMPTY)
								val += ",";
							val += el.options[j].value;
						}
					}
					break;
				}
				else if (el.type == "checkbox") {
					if (el.checked) {
						if (val != EMPTY)
							val += ",";
						val += el.value;
					}
				}
				else {
					alert("not supported type");
					return;
				}
			}

		}
	}
	return val;
}

function utilDisableInput(elem,disabled) {
	if (elem) {
		if (elem.length) {
			for (var i=0; i < elem.length; i++) {
				elem[i].disabled = disabled;
			}
		}
		else {
			elem.disabled = disabled;
		}
	}
}

function utilDisableOneOfElements(elems,value,disabled) {
	if (elems) {
		if (elems.length) {
			for (var i=0; i < elems.length; i++) {
				if (elems[i].value == value) {
					elems[i].disabled = disabled;
					break;
				}
			}
		}
		else {
			if (elems.value == value) {
				elems.disabled = disabled;
			}
		}
	}
}

function utilIsElementChecked(elems,value) {
	if (elems) {
		if (elems.length) {
			for (var i=0; i < elems.length; i++) {
				if (elems[i].value == value) {
					if (elems[i].checked && !elems[i].disabled)
						return SET;
					else
						return RESET;
				}
			}
		}
		else {
			if (elems.value == value) {
				if (elems.checked && !elems.disabled)
					return SET;
				else
					return RESET;
			}
		}
	}
	return NONE;
}

function utilExchangeValue(element1,element2)
{
	var value1,value2;
	if (element1 && element2) {
		value1 = element1.value;
		value2 = element2.value;

		element1.value = value2;
		element2.value = value1;
	}
}

function GetCheckedAuth(){

	}

function PermCheck(){
	var i=0;
	var j=0;
	
	if( !document.mainForm )
	{
		return;
	}
	
	for(i=0;i<document.mainForm.length;i++){
		if(document.mainForm.elements[i].name == "readableEntryIndicesIn"){
			if(document.mainForm.elements[i].checked && SELECTEDCHECK_LIST[j] == 0){
				SELECTEDCHECK_LIST[j] = 1;
				document.mainForm.elements[i+1].checked=false;
				document.mainForm.elements[i+2].checked=false;
				document.mainForm.elements[i+3].checked=false;
			}
			if(document.mainForm.elements[i+1].checked && SELECTEDCHECK_LIST[j+1] == 0){
				SELECTEDCHECK_LIST[j+1] = 1;
				document.mainForm.elements[i].checked=false;
				document.mainForm.elements[i+2].checked=false;
				document.mainForm.elements[i+3].checked=false;
			}
			if(document.mainForm.elements[i+2].checked && SELECTEDCHECK_LIST[j+2] == 0){
				SELECTEDCHECK_LIST[j+2] = 1;
				document.mainForm.elements[i].checked=false;
				document.mainForm.elements[i+1].checked=false;
				document.mainForm.elements[i+3].checked=false;
			}
			if(document.mainForm.elements[i+3].checked && SELECTEDCHECK_LIST[j+3] == 0){
				SELECTEDCHECK_LIST[j+3] = 1;
				document.mainForm.elements[i].checked=false;
				document.mainForm.elements[i+1].checked=false;
				document.mainForm.elements[i+2].checked=false;
			}

			if (document.mainForm.elements[i].checked == false && SELECTEDCHECK_LIST[j] == 1) {
				SELECTEDCHECK_LIST[j] = 0;
			}

			if (document.mainForm.elements[i+1].checked == false && SELECTEDCHECK_LIST[j+1] == 1) {
				SELECTEDCHECK_LIST[j+1] = 0;
			}

			if (document.mainForm.elements[i+2].checked == false && SELECTEDCHECK_LIST[j+2] == 1) {
				SELECTEDCHECK_LIST[j+2] = 0;
			}

			if (document.mainForm.elements[i+3].checked == false && SELECTEDCHECK_LIST[j+3] == 1) {
				SELECTEDCHECK_LIST[j+3] = 0;
			}
			j+=4;
		}
	}
}
function PermCheckForOwner(){
	var myIndexElemNum = 0;
	var ret = 0;

	myIndexElemNum = getMyIndexElemNum();
	ret = checkMyLineStat(myIndexElemNum);
	if(!ret){
		restoreMyLineStat(myIndexElemNum);
		return;
	}
	PermCheck();
	keepMyLineStat(myIndexElemNum);
}
function initKeepMyLineStat(){
	var myIndexElemNum = 0;

	myIndexElemNum = getMyIndexElemNum();
	if(!myIndexElemNum){
		return;
	}
	keepMyLineStat(myIndexElemNum);
}
function keepMyLineStat(myIndexElemNum){
	var i = 0;

	if( !document.mainForm )
	{
		return;
	}
	for(i=0; i<4; i++){
		if(document.mainForm.elements[myIndexElemNum + i].checked == true){
			ownerLineStatList[i] = 1;
		}
		else{
			ownerLineStatList[i] = 0;
		}
	}
}
function restoreMyLineStat(myIndexElemNum){
	var i = 0;

	if( !document.mainForm )
	{
		return;
	}
	for(i=0; i<4; i++){
		if(ownerLineStatList[i] == 1){
			document.mainForm.elements[myIndexElemNum + i].checked = true;
		}
		else{
			document.mainForm.elements[myIndexElemNum + i].checked = false;
		}
	}
}
function checkMyLineStat(myIndexElemNum){
	if( !document.mainForm )
	{
		return;
	}
	if(document.mainForm.elements[myIndexElemNum + 0].checked == false && document.mainForm.elements[myIndexElemNum + 1].checked == false && 
	   document.mainForm.elements[myIndexElemNum + 2].checked == false && document.mainForm.elements[myIndexElemNum + 3].checked == false)
	{
		return 0;
	}
	else{
		return 1;
	}
}
function getMyIndexElemNum(){
	var i = 0;

	if( !document.mainForm )
	{
		return;
	}
	for(i=0; i<document.mainForm.length; i++){
		if(document.mainForm.elements[i].name == "readableEntryIndicesIn" && document.mainForm.elements[i].id == "OWNER_R"){
			return i;
		}
	}
	return 0;
}

function getSkipInfo(skipname,key,skip_info) {
	var List;

	for (var i=0; i < SkipList.length; i++) {
		if (SkipList[i].skipname == skipname) {
			List = SkipList[i];

			skip_info.from	= getSkipUrlString(key,List.from);
			skip_info.via	= getSkipUrlString(key,List.via);
			skip_info.to	= getSkipUrlString(key,List.to);

			break;
		}
	}
}

function getSkipUrlString(key,urlInfo) {
	var url = EMPTY;
	var parts = new Array();
	parts[0] = urlInfo.cgi;
	parts[1] = urlInfo.marker;
	parts[2] = urlInfo.query;

	for (var i=0; i < parts.length; i++) {
		if (parts[i].length) {
			for (var j=0; j < parts[i].length; j++) {
				if (parts[i][j].key == key) {
					url += parts[i][j].value;
					break;
				}
			}
		}
		else {
			if( parts[i].value )
			{
				url += parts[i].value;
			}
		}
	}
	return url;
}

function adrsUndo() {
	var dest;
	if (BaseForm) {
		if (MainForm) {
			if (MainForm.wayFrom) {
				if (MainForm.wayFrom.value) {
					dest = MainForm.wayFrom.value;

					if (dest != EMPTY) {
						if (buttonCheckFlag) { return; }
						buttonCheckFlag = true;
						utilExchangeValue(MainForm.wayFrom,MainForm.wayTo);
						utilSubmitForm(MainForm,dest);
						utilPageSkipDisabled();
					}
					else {
						adrsListJump();
					}
				}
				else {
					adrsListJump();
				}
			}
			else {
				adrsListJump();
			}
		}
		else {
			adrsListJump();
		}
	}
}

function pageRefresh() {

	var el,item;
	var items = new refreshItemList();

	for (var i=0; i < MainForm.length; i++) {
		el = MainForm.elements[i];
		for (var j=0; j < items.item.length; j++) {
			item = items.item[j];
			if (item.arg.name && item.arg.value) {
				if (item.arg.name == el.name && item.arg.value == el.value) {
					el.disabled = item.disabled();
					break;
				}
			}
			else if (item.arg.name) {
				if (item.arg.name == el.name) {
					el.disabled = item.disabled();
					break;
				}
			}

			if (item.func) {
				item.func();
			}
		}
	}

	auth_protect_Refresh();
	changeFaxLineIn();
}

function setFaxLabel2In()
{
	var val = 0;

	val = utilGetIndexSelect(MainForm.faxOptionLabel2In);
	switch(val){
		case 1:
			MainForm.faxLabel2In.value = "";
			MainForm.faxLabel2In.disabled = SET;
			break;
		case 2:
			MainForm.faxLabel2In.disabled = RESET;
			break;
		default:
			MainForm.faxLabel2In.value = MainForm.faxOptionLabel2In.value;
			MainForm.faxLabel2In.disabled = RESET;
	}
}

function changeFaxLineIn()
{
	var result = false;
	var str;
	if(MainForm.faxLineIn){
		if(MainForm.faxLineIn.value == "CLOUD_FAX_O"){
			result = true;
		}
	}

	if(MainForm.faxDestIn) {
		if(result != false){
			MainForm.faxDestIn.maxLength = 30;
			str = MainForm.faxDestIn.value;
			MainForm.faxDestIn.value = str.substring(0,30);
		}
		else{
			MainForm.faxDestIn.maxLength = 128;
		}
	}
}

function auth_protect_Refresh() {

	var val,i;
	var protectSender,protectFolder
	var disp1,disp2;

	if((MainForm.entryUseIn)&&(MainForm.protectedUseIn)){

		for (i=0; i < MainForm.length; i++) {
			if (MainForm.elements[i].name == "protectedUseIn"){

				if (MainForm.elements[i].value == "PROTECT_SENDER_O"){
					protectSender = MainForm.elements[i];

				}else if (MainForm.elements[i].value == "PROTECT_FOLDER_O"){
					protectFolder = MainForm.elements[i];
				}
			}
		}

		val = utilGetInputValue(MainForm,"entryUseIn");
		if (val == "ENTRYUSE_TO_O"){
			if (protectSender){
				protectSender.disabled = SET;
			}
			if (protectFolder){
				protectFolder.disabled = CheckFolderEnable();
			}
		}else if(val == "ENTRYUSE_FROM_O"){
			if (protectSender){
				protectSender.disabled = RESET;
			}
			if (protectFolder){
				protectFolder.disabled = SET;
			}
		}else if(val == "ENTRYUSE_TO_O,ENTRYUSE_FROM_O"){
			if (protectSender){
				protectSender.disabled = RESET;
			}
			if (protectFolder){
				protectFolder.disabled = CheckFolderEnable();
			}
		}else{
			if (protectSender){
				protectSender.disabled = SET;
			}
			if (protectFolder){
				protectFolder.disabled = SET;
			}
		}

		if(MainForm.buttonProtectCode){
			if (protectSender){
				if((protectSender.disabled == RESET)&&(protectSender.checked == SET)){
					disp1 = SET;
				}else{
					disp1 = RESET;
				}
			}
			if (protectFolder){
				if((protectFolder.disabled == RESET)&&(protectFolder.checked == SET)){
					disp2 = SET;
				}else{
					disp2 = RESET;
				}
			}
			if ((disp1 == SET)||(disp2 == SET)){
				MainForm.buttonProtectCode.disabled = RESET;
			}else{
				MainForm.buttonProtectCode.disabled = SET;
			}
		}

	}else if(MainForm.protectedUseIn){
		if(MainForm.buttonProtectCode){
			if(MainForm.protectedUseIn.checked == SET){
				MainForm.buttonProtectCode.disabled = RESET;
			}else{
				MainForm.buttonProtectCode.disabled = SET;
			}
		}
	}
}
function CheckFolderEnable() {

	var folderProtocol = EMPTY;
	var servername = EMPTY;
	var pathname = EMPTY;

	if((MainForm.folderProtocolIn)&&(MainForm.folderServerNameIn)&&(MainForm.folderPathNameIn)){
		folderProtocol = utilGetInputValue(MainForm,"folderProtocolIn");
		servername = utilGetInputValue(MainForm,"folderServerNameIn");
		pathname = utilGetInputValue(MainForm,"folderPathNameIn");
		if(folderProtocol == "FTP_O"){
			if(servername != EMPTY){
				return RESET;
			}
		}else if((folderProtocol == "SMB_O")||(folderProtocol == "NCP_O")){
			if(pathname != EMPTY){
				return RESET;
			}
		}else{
			alert("unknown Protocol");
			alert(folderProtocol);
			return SET;
		}
	}
	return SET;
}

		MainForm.wayFrom.value = skip.from;
		MainForm.wayTo.value = skip.to;
		utilSubmitForm(MainForm,skip.via);
		utilPageSkipDisabled();
	}
}

function utilSubmitForm(form,dest,marker){
	var pos = 0;
	var queryString;
	var param,name,value;
	var queryPos = dest.indexOf("?");
	
	aplAvailSet();
	
	if (queryPos == -1) {
		form.action = dest;
	}
	else {
		form.action = dest.substring(0,queryPos);

		queryString = dest.substring(queryPos+1,dest.length);
		if (queryString.length > 0) {
			if (queryString.indexOf("&",queryString.length-1) == -1)
				queryString += "&";
		}

		while (pos < queryString.length) {
			param = queryString.substring(pos,queryString.indexOf("&",pos));
			pos += param.length + 1;

			if (param.indexOf("=") >= 0) {
				name = param.substring(0,param.indexOf("="));
				value = param.substring(param.indexOf("=")+1,param.length);
				for(var i=0; i < form.length; i++) {
					if (form.elements[i].name == name) {
						form.elements[i].value = value;
						break;
					}
				}
			}
		}
	}
	
	if( marker )
	{
		form.action += "#"+marker;
	}

	form.method = "post";
	form.target = "_self";
	formSubmit(form);
}

function WidthCheck(str)
{
	var w = 0;
	var maxlength = 0;
	var protocol = document.getElementsByName("folderProtocolIn")[0].value;

	if ((protocol == "SMB_O") || (protocol == "FTP_O")){
	
		for (var i=0; i<str.value.length; i++) {
			var c = str.value.charCodeAt(i);
			if ((c >= 0x0001 && 0x007e >= c) || (c >= 0xff60 && 0xff9f >= c)) {
				w++;
			}
			else {
				w+=2;
			}	
		}

		if (w > 255) {
			maxlength = i;
		}
		else{
			maxlength = 256;
		}

	}
	else
	{
		maxlength = 128;
	}

	$("input[name=folderPathNameIn]").attr("maxlength",maxlength);
}
