/* --------------------------------------------------------------- *
 * Init
 * --------------------------------------------------------------- */
function adrsWizardInit()
{
	/* 莽聰禄氓聝聫氓聢聺忙聹聼氓聦聳 */
	var idx = 0;
	
	for( idx = STEP_ID_BASE; idx <= IMG_ID_SELECT_STEP; idx++ )
	{
		/* 莽聰禄氓聝聫茫聛庐氓聟聢猫陋颅茫聛驴 */
		gl_stepImgArray[idx] = new Image();
		gl_stepImgArray[idx].src = IMAGE_PATH+"step"+idx+".gif";
		gl_stepImgArray[idx+IMG_ID_SELECT_STEP] = new Image();
		gl_stepImgArray[idx+IMG_ID_SELECT_STEP].src = IMAGE_PATH+"step"+idx+"select.gif";
	}
	
	/* scroll init */
	var divStep = _wizard_getStepDiv(STEP_ID_STEP);
	if( divStep.size() > 0 )
	{
		var mainModal = divStep.find("#frame_main-mdl");
		mainModal.scrollTop(0);
	}
	
	/* initialize global */
	gl_currentStep = STEP_ID_STEP;
	
	/* init item */
	folderProtocolChange();
	folderPasswordCheckChange();
	authPasswordCheckChange();
	authUserChange();
	
	/* pre input save */
	for( var stepId = STEP_ID_BASE; stepId <= STEP_ID_AUTH; stepId++ )
	{
		
		var inputArray = INPUT_NAME_ARRAY[stepId];
		jQuery.each( INPUT_NAME_ARRAY[stepId], function(inputIdx, inputName )
		{
			var confirmSpan = $(".span_"+inputName);
			gl_preInputArray[inputName] = new Array();
			confirmSpan.each(function(spanIndex)
			{
				gl_preInputArray[inputName][spanIndex] = $(this).html();
			});
		});
	}
}


/* --------------------------------------------------------------- *
 * Forward
 * --------------------------------------------------------------- */
function adrsWizardForward()
{
	var checkOk = true;
	switch( gl_currentStep )
	{
		case 0:/* 茫聜鹿茫聝聠茫聝聝茫聝聴茅聛赂忙聤聻莽聰禄茅聺垄	*/
			checkOk = _wizard_necessaryCheckStep();
			if( checkOk )
			{
				_wizard_clearMessage();
				_wizard_getStepListIn();
				_wizard_showStepImage();
				_wizard_confirmSelectStep();
				_wizard_stepForward();
			}
			else
			{
				_wizard_errorMessage( g_textRes_wizardNostepError );
			}
			break;
		case 1:/* 氓聼潞忙聹卢忙聝聟氓聽卤莽聰禄茅聺垄		*/
		case 2:/* 茫聝聲茫聜隆茫聝聝茫聜炉茫聜鹿氓庐聸氓聟聢	*/
		case 3:/* 茫聝隆茫聝录茫聝芦氓庐聸氓聟聢		*/
		case 4:/* 茫聝聲茫聜漏茫聝芦茫聝聙氓庐聸氓聟聢		*/
		case 5:/* 猫陋聧猫篓录忙聝聟氓聽卤			*/
			_wizard_aplAvailSet();
			checkOk = _wizard_necessaryCheck(gl_currentStep);
			if( checkOk )
			{
				_wizard_sendForm();
			}
			else
			{
				_wizard_errorMessage( g_textRes_wizardNecessaryError );
			}
			break;
		case 6:/* 莽垄潞猫陋聧莽聰禄茅聺垄			*/
			_wizard_finish();
			break;
		default:/* 茫聛聺茫聛庐盲禄聳	*/
			break;
	}
}

/* --------------------------------------------------------------- *
 * resize handler
 * --------------------------------------------------------------- */
function _wizard_resizeHandler(modalWidth,modalHeight)
{
	/* 茅聛赂忙聤聻 */
	for( stepId = STEP_ID_STEP; stepId <= STEP_ID_CONFIRM; stepId++ )
	{
		var divStep = _wizard_getStepDiv(stepId);
		if( divStep.size() > 0 )
		{
			var mainBox = divStep.find("#frame_main-mdl");
			var height = modalHeight - 174;/* 64(Bottom) + 98(Top) + 12(margin) */
			if( stepId != STEP_ID_STEP )
			{
				height -= 85;	/* step image */
			}
			if( height < 100 )
			{
				height = 100;
			}
			mainBox.height(height+"px");
		}
	}
}

/* --------------------------------------------------------------- *
 * step list in check
 * --------------------------------------------------------------- */
function _wizard_getStepListIn()
{
	var stepCnt = 0;
	gl_stepList = new Array();
	gl_stepList[stepCnt++] = STEP_ID_STEP;
	
	$("input[name='stepListIn']:checked").each(
	function()
	{
		var stepId = jQuery.inArray($(this).val(),STEP_STR_ARRAY);
		gl_stepList[stepCnt++] = stepId;
	});
	
	gl_stepList[stepCnt++] = STEP_ID_CONFIRM;
}


/* --------------------------------------------------------------- *
 * Show Step Image
 * --------------------------------------------------------------- */
function _wizard_showStepImage()
{
	/* 盲赂聙忙聶聜莽職聞茫聛芦氓聟篓茫聛娄茫聛庐莽聰禄氓聝聫茫聜聮茅聺聻猫隆篓莽陇潞茫聛芦茫聛聶茫聜聥 */
	$("#stepImage li").css("display","none");
	/* 盲赂聙忙聶聜莽職聞茫聛芦 step忙聲掳氓颅聴猫隆篓莽陇潞茫聜聮氓聣聤茅聶陇茫聛聶茫聜聥 */
	$(".wizardStepNumber").remove();
	
	/* 茅聛赂忙聤聻茫聛聲茫聜聦茫聛聼茫聜鹿茫聝聠茫聝聝茫聝聴茫聛芦氓聬聢茫聜聫茫聛聸茫聛娄莽聰禄茅聺垄茫聜聮忙聸麓忙聳掳茫聛聶茫聜聥 */
	jQuery.each(gl_stepList,function(idx,stepId)
	{
		if( stepId > STEP_ID_STEP )
		{
			/* 茫聜鹿茫聝聠茫聝聝茫聝聴茫聛庐忙聳聡氓颅聴氓聢聴氓聫聳氓戮聴 */
			var stepStr = STEP_STR_ARRAY[stepId];
			/* 猫隆篓莽陇潞茫聛聲茫聜聦茫聛娄茫聛聞茫聜聥茫聜驴茫聜陇茫聝聢茫聝芦茫聜聮氓聫聳氓戮聴 */
			var titleDiv = $("#step_"+stepStr+" .title");
			/* 茅聛赂忙聤聻茫聛聲茫聜聦茫聛聼茫聜鹿茫聝聠茫聝聝茫聝聴茫聜聮氓聫聳氓戮聴芒聡聮茫聜驴茫聜陇茫聝聢茫聝芦忙聸麓忙聳掳芒聡聮猫隆篓莽陇潞 */
			var stepImage = $("#stepImage"+idx);
			if( stepId == STEP_ID_CONFIRM )
			{
				stepImage.attr("class","non");
			}
			else
			{
				stepImage.removeAttr("class");
			}
			stepImage.find("p").html(titleDiv.html());
			/* 茫聜驴茫聜陇茫聝聢茫聝芦茫聛庐氓聣聧茫聛芦step忙聲掳茫聜聮猫隆篓莽陇潞茫聛聶茫聜聥 */
			titleDiv.prepend('<span class="wizardStepNumber">Step'+idx+'.</span>')
			/* 猫隆篓莽陇潞 */
			stepImage.css("display","block");
		}
	});
	
	/* 猫隆篓莽陇潞 */
	$("#stepImage").css("display","block");
}

/* --------------------------------------------------------------- *
 * Wizard Confirm Step Change
 * --------------------------------------------------------------- */
function _wizard_confirmSelectStep()
{
	/* 茅聛赂忙聤聻 */
	for( stepId = STEP_ID_BASE; stepId <= STEP_ID_AUTH; stepId++ )
	{
		var step_str = "confirm_"+STEP_STR_ARRAY[stepId];
		divStep = $("#"+step_str);
		if( divStep.size() > 0 )
		{
			var idx = jQuery.inArray(stepId,gl_stepList);
			var confirmCheck = divStep.find("#confirm_CHECK");
			var confirmUncheck = divStep.find("#confirm_UNCHECK");
			if( idx >= 0 )
			{
				confirmCheck.css("display","block");
				confirmUncheck.css("display","none");
			}
			/* 莽垄潞猫陋聧莽聰禄茅聺垄茫聛庐茫聜芦茫聝聠茫聜麓茫聝陋茫聜聮忙聹陋猫篓颅氓庐職茫聛芦茫聛聶茫聜聥 */
			else
			{
				confirmCheck.css("display","none");
				confirmUncheck.css("display","block");
			}
		}
	}
}

/* --------------------------------------------------------------- *
 * Step Forward
 * --------------------------------------------------------------- */
function _wizard_stepForward()
{
	var stepIndex = -1;
	/* 忙卢隆茫聛庐茫聜鹿茫聝聠茫聝聝茫聝聴茫聜聮忙陇聹莽麓垄 */
	var stepId;
	for( stepId = gl_currentStep + 1; stepId <= STEP_ID_CONFIRM; stepId++ )
	{
		stepIndex = jQuery.inArray(stepId,gl_stepList);
		if( stepIndex >= 0 )
		{
			break;
		}
	}
	/* 茫聝聲茫聜搂茫聝录茫聝芦茫聜禄茫聝录茫聝聲 */
	if( stepIndex < 0 )
	{
		stepId = STEP_ID_STEP;
		stepIndex = 0;
	}
	
	/* 莽垄潞猫陋聧莽聰禄茅聺垄莽聰聼忙聢聬 */
	if( STEP_ID_CONFIRM == stepId )
	{
		_wizard_createConfirm();
		_wizard_stepChange(stepId,stepIndex);
	}
	else
	{
		_wizard_stepChange(stepId,stepIndex);
		var step_str = STEP_STR_ARRAY[gl_currentStep];	
		var form = $("#form_"+step_str);
	}
	/* scroll init */
	var divStep = _wizard_getStepDiv(stepId);
	if( divStep.size() > 0 )
	{
		var mainModal = divStep.find("#frame_main-mdl");
		mainModal.scrollTop(0);
	}
}

/* --------------------------------------------------------------- *
 * Step Back
 * --------------------------------------------------------------- */
function _wizard_stepBack()
{
	/* 氓聣聧茫聛庐茫聜鹿茫聝聠茫聝聝茫聝聴茫聜聮忙陇聹莽麓垄 */
	var stepId;
	for( stepId = gl_currentStep - 1; stepId >= STEP_ID_STEP; stepId-- )
	{
		var stepIndex = jQuery.inArray(stepId,gl_stepList);
		if( stepIndex >= 0 )
		{
			break;
		}
	}
	/* 茫聝聲茫聜搂茫聝录茫聝芦茫聜禄茫聝录茫聝聲 */
	if( stepIndex < 0 )
	{
		stepId = STEP_ID_CONFIRM;
		stepIndex = 0;
	}
	
	if( stepId == STEP_ID_STEP )
	{
		_wizard_hiddenStepImage();
	}
	/* 忙聢禄茫聜聥 */
	_wizard_stepChange(stepId,stepIndex);
}
/* --------------------------------------------------------------- *
 * Step Change
 * --------------------------------------------------------------- */
function _wizard_stepChange(stepId,stepIdx)
{
	var oldStepIndex = jQuery.inArray(gl_currentStep,gl_stepList);
	
	_wizard_selectStepImage(oldStepIndex,false);
	_wizard_selectStepImage(stepIdx,true);
	
	_wizard_dispChange(gl_currentStep,stepId);
	
	gl_currentStep = stepId;
}

/* --------------------------------------------------------------- *
 * Disp Change
 * --------------------------------------------------------------- */
function _wizard_dispChange(oldStepId,newStepId)
{
	var oldStepDiv = _wizard_getStepDiv(oldStepId);
	var newStepDiv = _wizard_getStepDiv(newStepId);
	
	/* old茅聺聻猫隆篓莽陇潞芒聡聮new猫隆篓莽陇潞茫聛庐茅聽聠莽聲陋 */
	oldStepDiv.css("display","none");
	newStepDiv.css("display","block");
}

/* --------------------------------------------------------------- *
 * Select Step Image
 * --------------------------------------------------------------- */
function _wizard_selectStepImage(stepIdx,select)
{
	if( ( 0 < stepIdx )
	 && ( stepIdx < 7 )
	  )
	{
		var stepImage = $("#stepImage"+stepIdx+" img");
		if( select )
		{
			stepImage.attr("src",gl_stepImgArray[stepIdx+IMG_ID_SELECT_STEP].src);
		}
		else
		{
			stepImage.attr("src",gl_stepImgArray[stepIdx].src);
		}
	}
}

/* --------------------------------------------------------------- *
 * Send Form
 * --------------------------------------------------------------- */
function _wizard_sendForm()
{
	/* 忙漏聼氓聶篓茫聛聥茫聜聣茫聛庐氓驴聹莽颅聰氓戮聟茫聛隆 */
	/* 忙漏聼氓聶篓茫聛聥茫聜聣茫聛庐氓驴聹莽颅聰氓戮聟茫聛隆 */
	if( buttonCheckFlag ){return; }
	buttonCheckFlag = true;
	
	var step_str = STEP_STR_ARRAY[gl_currentStep];
	
	var form = $("#form_"+step_str);
	
	/* password encode */
	var passwordInput = form.find(":password");
	passwordInput.each(
		function()
		{
			var name_wk = $(this).attr("name");
			var password_wk = $(this).val();
			var name = name_wk.substr(3);
			var password = base64encode(password_wk);
			form.find("input[name='"+name+"']").val(password);
			$(this).attr("value_bk",password_wk);
			$(this).val("");
		});
	
	var postData = form.serialize();

	/* 茫聜篓茫聝漏茫聝录茫聜聮忙露聢茫聛聶 */
	var divStep = _wizard_getStepDiv(gl_currentStep);
	_wizard_clearMessage();
	removeImage( divStep, IMGPATH_ERROR );
	
	g_objModalDialog.wait(true);
	g_objConnectMachine.ajaxSend(
		ADRS_SET_USER_WIZARD,
		"POST",
		"mode="+g_wizardMode+"&step="+step_str+"&"+postData,
		_wizard_recvFunc);

	passwordInput.each(
		function()
		{
			$(this).val($(this).attr("value_bk"));
			$(this).attr("value_bk","");
		});
}

/* --------------------------------------------------------------- *
 * Recv Func
 * --------------------------------------------------------------- */
function _wizard_recvFunc(recvData)
{
	var str = "";
	var recvResult;
	
	var divStep = _wizard_getStepDiv(gl_currentStep);
	
	g_objModalDialog.wait(false);
	
	/* 茫聝聡茫聝录茫聜驴茫聝聛茫聜搂茫聝聝茫聜炉(氓聢聺茫聜聛茫聛聦"["茫聛陋茫聜聣莽碌聬忙聻聹) */
	if( recvData.indexOf("[") == 0 )
	{
		/* eval猫漏聲盲戮隆茫聛搂茫聜篓茫聝漏茫聝录茫聛聦氓聡潞茫聛聼氓聽麓氓聬聢茫聙聛catch茫聛搂茫聜篓茫聝漏茫聝录忙陇聹氓聡潞茫聛聶茫聜聥 */
		try
		{
			recvResult = eval(recvData);
		}
		catch(error)
		{
			_wizard_errorMessage( g_textRes_errorMessage );
			return;
		}
	}
	/* 茫聛聺茫聜聦盲禄楼氓陇聳茫聛炉茫聜篓茫聝漏茫聝录茫聝職茫聝录茫聜赂 */
	else
	{
		/* 忙漏聼氓聶篓茫聛庐氓驴聹莽颅聰氓戮聟茫聛隆茫聜聮猫搂拢茅聶陇茫聛聶茫聜聥 */
		g_objModalDialog.updateBodyOnly(recvData);
		return;
	}
	
	var resultId = new Number(recvResult[0]);

	/* 忙聢聬氓聤聼 */
	if( g_textResArray.length <= resultId )
	{
		_wizard_stepForward();
	}
	/* 氓陇卤忙聲聴 */
	else
	{
		/* 茫聜篓茫聝漏茫聝录茫聜聮氓聡潞茫聛聶 */
		_wizard_errorMessage( g_textResArray[resultId] );
		/* 茫聝聭茫聝漏茫聝隆茫聝录茫聜驴茫聜篓茫聝漏茫聝录 */
		if( recvResult.length > 1 )
		{
			var idx = 1;

function escapeHTML(str)
{
	return str
	.replace(/&/g, '&amp;')
	.replace(/</g, '&lt;')
	.replace(/>/g, '&gt;')
	.replace(/"/g, '&quot;')
	.replace(/'/g, '&#39;')
};

function _wizard_updateImageAppend(ddParent, preImage)
{

	var images = ddParent.find("img");
	var exist = false;
	images.each(function()
	{
		if( $(this).attr("src").indexOf(IMGPATH_UPDATE) >= 0 )
		{
			exist = true;
		}
	});
	if( exist )
	{
		return;
	}

	{
		ddParent.find("span:eq(2)").after(updateImg);
	}
	else
	{
		ddParent.append(updateImg);
	}
}

function itemCOPYCheck(n)
{
	var i=0;
	var j=0;
	var form = document.getElementById("form_AUTH");
	if( !form )
	{
		return;
	}
	if( !form.availableFuncIn[n-1] )
	{
		return;
	}
	if(form.availableFuncIn[n-1].checked==true){
		if (form.availableFuncIn[n-1].value=="COPY_FC") {
			if (form.acsLimitationIn) {
				for(i=0;i<form.acsLimitationIn.length;i++){
					form.acsLimitationIn[i].disabled=false;
				}
			}
	    }
		for(i=n-1;i<form.availableFuncIn.length;i++){
			if (form.availableFuncIn[i].value.indexOf("COPY_")>=0) {
				form.availableFuncIn[i].checked=true;
		    }
		}
	}else{
		for(i=n-1;i>=0;i--){
			if (form.availableFuncIn[i].value.indexOf("COPY_")>=0) {
				form.availableFuncIn[i].checked=false;
				if (form.availableFuncIn[i].value=="COPY_FC") {
					if (form.acsLimitationIn) {
						for(j=0;j<form.acsLimitationIn.length;j++){
							form.acsLimitationIn[j].disabled=true;
						}
					}
			    }
		    }
		}
	}
}
